country_name.csv =  reported weekly Zika cases by country

Brazil_*.csv = reported weekly Zika cases by Brazilian region

TimeWindows*.csv =  time windows for estimation of reproduction number used by R code

R_estimates_*.csv = reproduction number estimates for Latin American countries and Brazilian regions

*.R = R code for reproduction number estimation (needs EpiEstim package)

zika spatial_release.mmd = Berkeley Madonna code for stochastic metapopulation transmission dynamic model of Zika (undocumented - for completeness only)
